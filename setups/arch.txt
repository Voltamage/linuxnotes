download arch iso from here, use the magnet link
    https://www.archlinux.org/download/
    
verify your file with these commands, replace filename with current arch version
make the signature files by copy pasting the signature and then a space and then the filename
    gpg --keyserver-options auto-key-retrieve --verify archlinux-2019.08.01-x86_64.iso.sig
    md5sum -c archlinux-2019.08.01-x86_64.iso.md5
    sha1sum -c archlinux-2019.08.01-x86_64.iso.sha1
    
burn usb with etcher, as it seems to handle uefi successfully
boot into uefi mode

you can read the install guide with
    nano -w install.txt
    
check uefi mode
    ls /sys/firmware/efi/efivars
    
check network interfaces and connect
    ip link
    dhcpcd
    
test internet connection
    ping www.archlinux.org -c 3
    
sync and check system time
    timedatectl set-ntp true
    timedatectl status
    
list drives and partitions
    lsblk
    
destroy tables
    gdisk
    x
    z

partition with cfdisk
    cfdisk /dev/sda
select gpt table
new partition, 512M size, EFI System
new partition, 16G size, Linux swap
new partition, 32G size, Linux filesystem
new partition, ooG size, Linux filesystem

check alignment
    parted -a optimal /dev/sda
    align-check opt 1
    align-check opt 2
    align-check opt 3
    align-check opt 4

format filesystem
    lsblk
    mkfs.fat -F32 /dev/sda1
    mkswap /dev/sda2
    mkfs.ext4 /dev/sda3
    mkfs.ext4 /dev/sda4
    
check partitions
    lsblk
    lsblk -f
    
mount filesystem
    swapon /dev/sda2
    mount  /dev/sda3 /mnt
    mkdir /mnt/boot
    mkdir /mnt/home
    mount  /dev/sda1 /mnt/boot
    mount  /dev/sda4 /mnt/home
    
install arch
    pacstrap /mnt base base-devel
    
generate fstab and check it
    genfstab -U /mnt >> /mnt/etc/fstab
    nano -w /mnt/etc/fstab

enter arch
    arch-chroot /mnt
    
list and set timezone
    ls usr/share/zoneinfo
    ls usr/share/zoneinfo/America
    ln -sf /usr/share/zoneinfo/America/Los_Angeles /etc/localtime
    hwclock --systohc
    
localization, uncomment en_US.UTF-8 UTF-8
    nano -w /etc/locale.gen
    locale-gen
    echo LANG=en_US.UTF-8 > /etc/locale.conf
    nano -w /etc/locale.conf
    
hosts
    echo arch > /etc/hostname
    nano -w /etc/hosts
        127.0.0.1	localhost
        ::1 		localhost
        127.0.1.1	arch.localdomain	arch
        
rebuild initramfs just in case
    mkinitcpio -p linux
    
set root password
    passwd
    
efi, grub, and bootloader
    pacman -S efibootmgr grub os-prober
    mkdir /boot/efi
    
mirrors
    pacman -S reflector
    cp /etc/pacman.d/mirrorlist /etc/pacman.d/mirrorlist.backup
    reflector --latest 32 --protocol https --sort rate --save /etc/pacman.d/mirrorlist
    nano -w /etc/pacman.d/mirrorlist
    nano -w /etc/systemd/system/reflector.service
        [Unit]
        Description=mirrorlist update
        Wants=network-online.target
        After=network-online.target
        
        [Service]
        Type=oneshot
        ExecStart=/usr/bin/reflector --latest 32 --protocol https --sort rate --save /etc/pacman.d/mirrorlist
        
        [Install]
        RequiredBy=multi-user.target
        
systemctl enable dhcpcd
timedatectl set-ntp true
pacman -S grub os-prober intel-ucode
grub-install --recheck --target=i386-pc /dev/sdX
grub-mkconfig -o /boot/grub/grub.cfg
exit
umount -R /mnt
reboot
